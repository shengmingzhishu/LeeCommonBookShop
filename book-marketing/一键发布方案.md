## 二、发布方式对比

### 方案一：官方API（推荐）

**优点**：
- 官方支持，稳定可靠
- 可批量操作，效率高
- 支持定时发布

**缺点**：
- 需要申请开发者权限
- 各平台API不同，需分别对接

**实现方式**：
1. 申请各平台开发者账号
2. 获取API密钥
3. 编写发布脚本

### 方案二：第三方分发平台

**常见平台**：
- 蚁小二
- 易撰
- 新榜
- 乐观号

**优点**：
- 一键分发多平台
- 不需要开发

**缺点**：
- 需要付费
- 可能有内容限制
- 依赖第三方服务

### 方案三：浏览器自动化

**优点**：
- 不需要API权限
- 模拟人工操作

**缺点**：
- 不稳定，容易被检测
- 维护成本高

### 3.3 知乎发布方案

知乎有两种发布方式：

#### 方式一：知乎专栏API（官方）

```javascript
// 知乎专栏 API（需要登录获取token）
// https://zhuanlan.zhihu.com/api/columns/{columnId}/articles

POST /api/columns/{columnId}/articles
Headers: {
  "x-xsrftoken": "token",
  "x-zse-83": "加密参数"
}
Body: {
  "title": "文章标题",
  "content": "文章内容（Markdown/HTML）",
  "topicUrls": ["话题URL"],
  "canComment": true
}
```

**难点**：知乎API需要登录态，token有效期有限

#### 方式二：浏览器自动化（推荐）

使用 agent-browser 技能模拟登录发布：

```javascript
// zhihu-publish.js

const { chromium } = require('playwright');

async function publishToZhihu(article) {
  const browser = await chromium.launch({ headless: false });
  const context = await browser.newContext();
  
  // 使用保存的登录态
  await context.addCookies(require('./zhihu-cookies.json'));
  
  const page = await context.newPage();
  
  // 进入专栏管理页
  await page.goto('https://zhuanlan.zhihu.com/write');
  
  // 点击新建文章
  await page.click('[data-zop-feedlist="new_article"]');
  
  // 填写标题
  await page.fill('.TitleInput input', article.title);
  
  // 填写内容（Markdown编辑器）
  await page.fill('.RichTextInput', article.content);
  
  // 添加话题
  for (const topic of article.topics) {
    await page.click('.TopicSelector');
    await page.fill('.TopicSelector input', topic);
    await page.press('.TopicSelector input', 'Enter');
  }
  
  // 发布
  await page.click('[data-zop-feedlist="publish"]');
  
  await browser.close();
}
```

## 四、实施步骤

### 第一步：申请平台权限

1. 注册头条号、大鱼号账号
2. 申请开发者权限
3. 获取API密钥

### 第二步：开发发布脚本

```javascript
// publish.js 示例

const platforms = {
  toutiao: require('./platforms/toutiao'),
  dayu: require('./platforms/dayu'),
  baijia: require('./platforms/baijia')
};

async function publish(article, selectedPlatforms) {
  const results = {};
  
  for (const platform of selectedPlatforms) {
    try {
      const result = await platforms[platform].publish(article);
      results[platform] = { success: true, url: result.url };
    } catch (error) {
      results[platform] = { success: false, error: error.message };
    }
  }
  
  return results;
}

// 使用示例
const article = {
  title: '三年没升职，领导终于说了实话',
  content: '...',
  cover: 'cover.jpg',
  tags: ['职场', '升职']
};

publish(article, ['toutiao', 'dayu']);
```

### 第三步：批量发布

```javascript
// batch-publish.js

const fs = require('fs');
const path = require('path');

async function batchPublish(articlesDir, platforms) {
  const articles = fs.readdirSync(articlesDir);
  
  for (const file of articles) {
    const article = parseArticle(path.join(articlesDir, file));
    await publish(article, platforms);
    console.log(`已发布: ${article.title}`);
  }
}
```

## 五、内容适配

不同平台有不同要求，需要适配：

| 平台 | 标题长度 | 内容格式 | 封面要求 |
|------|----------|----------|----------|
| 今日头条 | 5-30字 | HTML/Markdown | 1-3张 |
| 大鱼号 | 5-30字 | HTML | 1张 |
| 百家号 | 5-30字 | HTML | 1张 |
| 公众号 | 无限制 | HTML | 1张 |

**适配函数**：
```javascript
function adaptArticle(article, platform) {
  return {
    title: article.title.slice(0, 30),
    content: convertToHtml(article.content),
    cover: article.covers[0],
    tags: article.tags.slice(0, 5)
  };
}
```
